117 template <typename T, typename... Args>
118 requires std::constructible_from<T, Args...>
119 inline void AddComponent(Entity::ID id, Args&&... args)
120 {
121 if (masks.find(id) == masks.end())
122 throw std::runtime_error("Invalid Entity ID");
123
124 Component::Tag tag = GetTag<T>();
125 if (masks[id][tag])
126 throw std::runtime_error("Component already exists for this Entity");
127
128 masks[id].set(tag);
129
130 if(pools.find(tag) == pools.end())
131 pools[tag] = std::make_shared<ComponentPool<T>>();
132
133 auto pool = std::static_pointer_cast<ComponentPool<T>>(pools[tag]);
134 auto component = pool->Acquire(std::forward<Args>(args)...);
135 components[id][tag] = std::move(component);
136 }

147 template <typename T>
148 inline T* GetComponent(Entity::ID id)
149 {
150 Component::Tag tag = GetTag<T>();
151 if(!masks[id][tag])
152 return nullptr;
153
154 auto it = components.find(id);
155
156 if(it != components.end() && it->second.find(tag) != it->second.end())
157 return static_cast<T*>(it->second[tag].get());
158
159 return nullptr;
160 }
