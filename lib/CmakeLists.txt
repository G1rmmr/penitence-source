file(GLOB_RECURSE
    SRC_FILES
    CONFIGURE_DEPENDS
    ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

# GAlpha
add_library(GAlpha STATIC ${SRC_FILES})

# Set SDL2
find_package(SDL2 2.26.12 REQUIRED)
find_package(SDL2_image 2.6.3 REQUIRED)

# Include dir
target_include_directories(GAlpha PUBLIC
    ${CMAKE_SOURCE_DIR}/includes
    ${SDL2_INCLUDE_DIRS}
    ${SDL2_IMAGE_INCLUDE_DIRS}
)

# Link libs
target_link_libraries(GAlpha PUBLIC
    SDL2::SDL2main
    SDL2::SDL2
    SDL2_image::SDL2_image
)

# Compile opt
target_compile_options(GAlpha PRIVATE -Wall -Werror)
